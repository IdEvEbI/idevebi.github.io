# CentOS 维护笔记

购置华为云虚拟主机一台，配置如下：2 CPU / 4GB / CentOS 7.5 64bit。

本文用于记录虚拟主机的工具安装，以便速查。

# 一. 常用工具安装

  1. 系统升级

    yum update && yum upgrade

# 列出所有已安装的软件包
yum list installed

  2. Git

    yum install -y git

  3. Vim 8.0

    rpm -Uvh http://mirror.ghettoforge.org/distributions/gf/gf-release-latest.gf.el7.noarch.rpm
rpm --import http://mirror.ghettoforge.org/distributions/gf/RPM-GPG-KEY-gf.el7

yum -y remove vim-minimal vim-common vim-enhanced
yum -y --enablerepo=gf-plus install vim-enhanced sudo

  4. net-tools

    yum install -y net-tools

  5. wget

    yum install -y wget

# 二. Node.js

  1. 添加 Node.js 的 yum 源

    # LTS 版本
curl -sL https://rpm.nodesource.com/setup_12.x | sudo bash -

  2. 安装

    yum install -y nodejs

  3. 确认安装

    node -v

npm -v

# 三. MySQL 5.7

MySQL 的 glibc 版是由 MySQL 提供的经编译之后的软件包，在 CentOS 上安装比较简单。

  1. 准备工作

    # 1. 卸载 `mariadb-libs` 软件包
yum remove mariadb-libs

# 2. 安装 `libaio`
yum install -y libaio

# 3. 创建特殊的账号：`mysql`（所属组`mysql`，不允许登录）
useradd -r -s /sbin/nologin mysql

  2. 下载并安装 MySQL 5.7 的 glibc 版本

    # 1. 获取 5.7 版本的安装文件（使用断点续传方式下载）
wget -c https://cdn.mysql.com//Downloads/MySQL-5.7/mysql-5.7.32-linux-glibc2.12-x86_64.tar.gz

# 2. 解压缩安装包
tar zxvf mysql-5.7.32-linux-glibc2.12-x86_64.tar.gz

# 3. 将 mysql 文件移动到 `/usr/local/mysql` 目录
mv mysql-5.7.32-linux-glibc2.12-x86_64 /usr/local/mysql

# 4. 修改 `/usr/local/mysql` 目录权限
chown mysql:mysql /usr/local/mysql/

  3. 启动 MySQL 服务

    # 1. 切换目录
cd /usr/local/mysql/

# 2. 初始化数据库（提示：初始化之后会生成一个临时密码，**这个密码务必要记住**）
bin/mysqld --initialize --user=mysql

# 3. 复制 MySQL 数据库服务文件
cp support-files/mysql.server /etc/init.d/mysql

# 4. 将 MySQL 数据库服务设置为自启动服务（`disable` 关闭自启动）
systemctl enable mysql

# 5. 启动 MySQL 数据库服务（`stop` 停止服务 / `status` 查看服务状态）
systemctl start mysql

# 6. 更新 `.bash_profile` 设置 `$PATH` 环境变量
echo -e '\n# mysql\nexport PATH="/usr/local/mysql/bin:$PATH"' >> ~/.bash_profile

    TIP

    初始化之后会生成一个临时密码，这个密码务必要记住。

  4. 修改初始密码

    WARNING

    初始安装之后，必须要先修改密码，否则其他操作都执行不了。

    1. 登录 MySQL

      mysql -uroot -p

    2. 修改密码

      -- 1. 修改密码
mysql> alter user root@localhost identified by 'passwd';

-- 2. 退出
mysql> quit

    3. 再次登录 MySQL 确认密码生效

  5. 设置防火墙

    WARNING

    出于安全性的考虑，MySQL 的初始安装之后，是不允许被远程访问的。

    # 启动防火墙
systemctl start firewalld

# 将 firewalld 设置为自启动服务
systemctl enable firewalld

# 添加 `3306` 端口，参数说明：
# --zone 作用域
# --add-port 添加端口（端口/通讯协议）
# --permanent 永久生效，没有此参数重启后失效
firewall-cmd --zone=public --add-port=3306/tcp --permanent

# 重启防火墙
firewall-cmd --reload

# 查看防火墙状态
firewall-cmd --state

# 查看防火墙端口列表
firewall-cmd --list-ports

  6. 允许 root 远程访问

    -- 1. 切换数据库
mysql> use mysql;

-- 2. 允许远程访问
mysql> GRANT ALL PRIVILEGES ON *.* TO 'root'@'%' IDENTIFIED BY 'PASSWORD' WITH GRANT OPTION;

-- 3. 刷新特权
mysql> FLUSH PRIVILEGES;

-- 4. 退出
mysql> quit

  7. 重启 MySQL

    systemctl restart mysql

在 CentOS 8 安装 MySQL 5.7 的教程可参考：https://computingforgeeks.com/install-mysql-5-7-on-centos-rhel-linux/ (opens new window)。

# 四. Redis 6.x

  1. 允许 Remi 仓库

    yum install epel-release yum-utils
yum install http://rpms.remirepo.net/enterprise/remi-release-7.rpm
yum-config-manager --enable remi

  2. 安装 Redis

    yum install redis

  3. 启动服务

    systemctl start redis
systemctl enable redis

  4. 检查服务状态

    systemctl status redis
